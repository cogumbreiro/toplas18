Control-flow reduction:

Module ControlFlow.
Inductive Reduces : cflow -> prog -> prog -> Prop :=
  | r_skip:
    forall p,
    Reduces skip p p
  | r_iter:
    forall p q,
    Reduces (loop p) q (concat p (LOOP(p);; q))
  | r_elide:
    forall p q,
    Reduces (loop p) q q.
End ControlFlow.
